Example 1

%{
create two random vector A and B then use a for loop to create a vector C
such that:
> first element of C is sum of firstelement of A and last element of B
> and second element of C is sum of second element of A and second
  last element of B
%}
clc; clear;
N = 10;
A = randi(100,N,1);
B = randi(100,N,1);   % creating a B matrix of 10X1
C = zeros(N,1);
for i = 1:N
    C(i,1) = A(i,1)+B(N+1-i,1);
end
disp(A);
disp(B);
disp(C);

Example 2

%{
create and plot mathematical function f = A*sin(t)+B*exp(5*t) using while
loop
%}
clc; clear;
t = 0:2:100;
A = 2;
B = 0.25;
k = 1;
while k < length(t)+1
    f = A*sin(t(k))+B*exp(5*t(k));
    figure(1)
    plot(t(k),f,'r-o')
    hold on
    k = k+1;
end
xlabel('time in sec')
ylabel('function f')
title('Plot of f')
% axis([0 5 0 15])
hold off

Example 3

%{
    Use if else statement to check the which weekday fall on user input
    day of year 2023
%}
clc; clear;
U = input('Enter the Day from 1 to 365 N :'); % get the input from user
Rem = mod(U,7); %  calculate remainder of input U
if Rem == 1
    fprintf('Day is Sunday')
elseif Rem == 2
    fprintf('Day is Monday')
elseif Rem == 3
    fprintf('Day is Tuesday')
elseif Rem == 4
    fprintf('Day is Wednesday')
elseif Rem == 5
    fprintf('Day is Thursday')
elseif Rem == 6
    fprintf('Day is Friday')
else
    fprintf('Day is Saturday')
end

Example 5

%{
find the root of equation x^3-5 using bisection method
create a function file of bisection.m
%}
clc; clear;
e = 1e-3;  % permissible error
f = @(x)x^3-5; % Anonymous function to define equation
x_star = bisection(e,f,1,2); % calling function to find root of equation
disp(x_star) % display root in command windown

Example 6

%{
    Use of diag command
%}

clc; clear;
N = 30;
step_size = 0.03;

% xvector
x = 0:step_size:1;

% creating  d= main diagonal, u= upper diagonal l= lower diagonal
d = -2*ones(N,1);
u = 1*ones(N-1,1);
l = 1*ones(N-1,1);

% Matrix A of size NxN and vector B of size Nx1, respectively
A = diag(d,0)+diag(l,-1)+diag(u,1);
B = zeros(N,1);
B(1,1) = -1;
B(N,1)= -exp(1);

% Numerical solution
y = A\B;
